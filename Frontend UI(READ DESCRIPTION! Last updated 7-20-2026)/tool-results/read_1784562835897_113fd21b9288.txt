   188Ôćĺ/**
   189Ôćĺ * Append a signed smart-encoded value to a byte array.
   190Ôćĺ *
   191Ôćĺ * Matches the format read by `Packet.gsmart()`:
   192Ôćĺ *   - value in [-64, 63]        -> 1 byte  = value + 0x40
   193Ôćĺ *   - value in [-16384, 16383]  -> 2 bytes = value + 0xc000 (big-endian)
   194Ôćĺ *
   195Ôćĺ * Local copy of the helper in `OsrsModel.ts` / `OsrsAnimFrame.ts` ÔÇö used
   196Ôćĺ * here to emit the model's vertex-delta and triangle-index streams. If
   197Ôćĺ * this proves generally useful, future task should hoist it into
   198Ôćĺ * `Packet.ts` as `psmartSigned()` (per Task 5-a's next-action hook).
   199Ôćĺ */
   200Ôćĺfunction psmartSigned(out: number[], value: number): void {
   201Ôćĺ    if (value >= -64 && value <= 63) {
   202Ôćĺ        out.push((value + 0x40) & 0xff);
   203Ôćĺ    } else if (value >= -16384 && value <= 16383) {
   204Ôćĺ        const v: number = (value + 0xc000) & 0xffff;
   205Ôćĺ        out.push((v >> 8) & 0xff);
   206Ôćĺ        out.push(v & 0xff);
   207Ôćĺ    } else {
   208Ôćĺ        throw new Error(`psmartSigned: value ${value} out of smart range [-16384, 16383]`);
   209Ôćĺ    }
   210Ôćĺ}

   212Ôćĺ/**
   213Ôćĺ * Synthesize a minimal but valid OSRS model blob.
   214Ôćĺ *
   215Ôćĺ * Layout (matching `OsrsModel.decode` byte-for-byte):
   216Ôćĺ *
   217Ôćĺ *   body:
   218Ôćĺ *     [vertex flags]     4 bytes (all 0 ÔÇö no deltas, all vertices at origin)
   219Ôćĺ *     [vertex X]         0 bytes (all deltas 0 Ôćĺ no X bytes emitted)
   220Ôćĺ *     [vertex Y]         0 bytes
   221Ôćĺ *     [vertex Z]         0 bytes
   222Ôćĺ *     [vertex labels]    absent (hasVertexLabels = 0)
   223Ôćĺ *     [triangle orient]  2 bytes ([1, 1] ÔÇö both triangles are type 1)
   224Ôćĺ *     [triangle colours] 4 bytes (2 triangles ├Ś u16 each ÔÇö colour 0x1234)
   225Ôćĺ *     [triangle infos]   absent (hasFaceColors = 0)
   226Ôćĺ *     [triangle materials] absent
   227Ôćĺ *     [triangle alphas]  absent
   228Ôćĺ *     [triangle tags]    absent
   229Ôćĺ *     [triangle priorities] absent (priority = 0, not 255)
   230Ôćĺ *     [triangle verts]   6 bytes (3 smart-encoded deltas per type-1 triangle)
   231Ôćĺ *     [texture UVs]      absent (triangleTextureCount = 0)
   232Ôćĺ *     [triangle skins]   absent
   233Ôćĺ *     [vertex skins]     absent
   234Ôćĺ *   trailer (23 bytes):
   235Ôćĺ *     version=1, vertexCount=4, triangleCount=2, triangleTextureCount=0,
   236Ôćĺ *     hasFaceColors=0, hasFaceTextures=0, priority=0, hasFaceAlphas=0,
   237Ôćĺ *     hasTriangleTags=0, hasVertexLabels=0,
   238Ôćĺ *     dataLenVertexX=0, dataLenVertexY=0, dataLenVertexZ=0,
   239Ôćĺ *     dataLenFaceOrientations=6, triangleSkinCount=0, vertexSkinCount=0,
   240Ôćĺ *     maxDepth=0
   241Ôćĺ *
   242Ôćĺ * Triangle vertex indices (type-1 encoding, 3 deltas per triangle):
   243Ôćĺ *   triangle 0: vertices (0, 1, 2). With last=0 (initial):
   244Ôćĺ *     delta_a = 0-0 = 0  Ôćĺ smart 0x40
   245Ôćĺ *     delta_b = 1-0 = 1  Ôćĺ smart 0x41
   246Ôćĺ *     delta_c = 2-1 = 1  Ôćĺ smart 0x41
   247Ôćĺ *     last = 2
   248Ôćĺ *   triangle 1: vertices (0, 2, 3). With last=2:
   249Ôćĺ *     delta_a = 0-2 = -2 Ôćĺ smart 0x3e
   250Ôćĺ *     delta_b = 2-0 = 2  Ôćĺ smart 0x42
   251Ôćĺ *     delta_c = 3-2 = 1  Ôćĺ smart 0x41
   252Ôćĺ *
   253Ôćĺ * Total body: 4 + 0 + 0 + 0 + 0 + 2 + 4 + 6 = 16 bytes.
   254Ôćĺ * Total blob: 16 + 23 = 39 bytes.
   255Ôćĺ */
   256Ôćĺfunction synthesizeOsrsModelBlob(): Uint8Array {
   257Ôćĺ    const body: number[] = [];
   258Ôćĺ
   259Ôćĺ    // 1. vertex flags (4 bytes, all 0 ÔÇö no deltas)
   260Ôćĺ    body.push(0, 0, 0, 0);
   261Ôćĺ
   262Ôćĺ    // 2-4. vertex X/Y/Z ÔÇö omitted (all deltas are 0).
   263Ôćĺ
   264Ôćĺ    // 5. vertex labels ÔÇö omitted (hasVertexLabels = 0).
   265Ôćĺ
   266Ôćĺ    // 5.5 triangle orientation bytes (2 bytes ÔÇö both type 1)
   267Ôćĺ    body.push(1, 1);
   268Ôćĺ
   269Ôćĺ    // 6. triangle colours (2 ├Ś u16 = 4 bytes) ÔÇö colour 0x1234 each.
   270Ôćĺ    body.push(0x12, 0x34);
   271Ôćĺ    body.push(0x12, 0x34);
   272Ôćĺ
   273Ôćĺ    // 7-10.5 optional sections ÔÇö all omitted (flags are 0).
   274Ôćĺ
   275Ôćĺ    // 11. triangle vertex indices (smart-encoded deltas)
   276Ôćĺ    // triangle 0: deltas (0, 1, 1)
   277Ôćĺ    psmartSigned(body, 0);
   278Ôćĺ    psmartSigned(body, 1);
   279Ôćĺ    psmartSigned(body, 1);
   280Ôćĺ    // triangle 1: deltas (-2, 2, 1)
   281Ôćĺ    psmartSigned(body, -2);
   282Ôćĺ    psmartSigned(body, 2);
   283Ôćĺ    psmartSigned(body, 1);
   284Ôćĺ
   285Ôćĺ    const dataLenFaceOrientations = 6; // 3 bytes ├Ś 2 triangles
   286Ôćĺ    const dataLenX = 0;
   287Ôćĺ    const dataLenY = 0;
   288Ôćĺ    const dataLenZ = 0;
   289Ôćĺ
   290Ôćĺ    // 12-14. optional sections ÔÇö omitted.
   291Ôćĺ
   292Ôćĺ    // ---- 23-byte OSRS trailer ----
   293Ôćĺ    const trailer: number[] = [];
   294Ôćĺ    trailer.push(1);           // version (>= 1 = OSRS)
   295Ôćĺ    trailer.push(0, 4);        // vertexCount = 4 (u16 BE)
   296Ôćĺ    trailer.push(0, 2);        // triangleCount = 2 (u16 BE)
   297Ôćĺ    trailer.push(0);           // triangleTextureCount = 0
   298Ôćĺ    trailer.push(0);           // hasFaceColors = 0
   299Ôćĺ    trailer.push(0);           // hasFaceTextures = 0
   300Ôćĺ    trailer.push(0);           // priority = 0 (shared; not 255 = no per-face)
   301Ôćĺ    trailer.push(0);           // hasFaceAlphas = 0
   302Ôćĺ    trailer.push(0);           // hasTriangleTags = 0
   303Ôćĺ    trailer.push(0);           // hasVertexLabels = 0
   304Ôćĺ    trailer.push((dataLenX >> 8) & 0xff, dataLenX & 0xff);
   305Ôćĺ    trailer.push((dataLenY >> 8) & 0xff, dataLenY & 0xff);
   306Ôćĺ    trailer.push((dataLenZ >> 8) & 0xff, dataLenZ & 0xff);
   307Ôćĺ    trailer.push((dataLenFaceOrientations >> 8) & 0xff, dataLenFaceOrientations & 0xff);
   308Ôćĺ    trailer.push(0);           // triangleSkinCount = 0
   309Ôćĺ    trailer.push(0);           // vertexSkinCount = 0
   310Ôćĺ    trailer.push(0);           // maxDepth = 0
   311Ôćĺ
   312Ôćĺ    const total = body.length + trailer.length;
   313Ôćĺ    const out = new Packet(new Uint8Array(total));
   314Ôćĺ    for (const b of body) {
   315Ôćĺ        out.p1(b);
   316Ôćĺ    }
   317Ôćĺ    for (const b of trailer) {
   318Ôćĺ        out.p1(b);
   319Ôćĺ    }
   320Ôćĺ    return out.data;
   321Ôćĺ}

   323Ôćĺ/**
   324Ôćĺ * Synthesize a minimal OSRS AnimBase (skeleton) blob.
   325Ôćĺ *
   326Ôćĺ * Layout (byte-identical to 377 skeleton format):
   327Ôćĺ *   length: u8 = 5
   328Ôćĺ *   types:  u8 ├Ś 5 = [0, 1, 2, 3, 5]  (OP_BASE, OP_TRANSLATE, OP_ROTATE, OP_SCALE, OP_ALPHA)

   344Ôćĺ/**
   345Ôćĺ * Synthesize a minimal OSRS anim frame blob.
   346Ôćĺ *
   347Ôćĺ * Layout (matching `OsrsAnimFrame.decode` byte-for-byte):
   348Ôćĺ *
   349Ôćĺ *   body (before trailing 2-byte base ID):
   350Ôćĺ *     u16 frameLength          = 5
   351Ôćĺ *     per group 0..4:
   352Ôćĺ *       u8 flags               (group 0: 0x7 = X+Y+Z; group 1: 0x1 = X; others: 0)
   353Ôćĺ *       smart X (if flag & 0x1)
   354Ôćĺ *       smart Y (if flag & 0x2)
   355Ôćĺ *       smart Z (if flag & 0x4)
   356Ôćĺ *     u8 delay                 = 2
   357Ôćĺ *   trailer (2 bytes):
   358Ôćĺ *     u16 baseId
   359Ôćĺ *
   360Ôćĺ * For frame 2005 (stand):
   361Ôćĺ *   group 0 (OP_BASE): flags=0x7, X=10, Y=20, Z=30 (3 smart bytes)
   362Ôćĺ *   group 1 (OP_TRANSLATE): flags=0x1, X=5 (1 smart byte)
   363Ôćĺ *   groups 2,3,4: flags=0
   364Ôćĺ *
   365Ôćĺ * For frame 2006 (attack):
   366Ôćĺ *   group 0 (OP_BASE): flags=0x7, X=0, Y=0, Z=0 (3 smart bytes)
   367Ôćĺ *   groups 1,2,3,4: flags=0
   368Ôćĺ *
   369Ôćĺ * Total blob: 2 (frameLength) + (1+3) + (1+1) + 1 + 1 + 1 (3 zero-flag groups) + 1 (delay) + 2 (baseId) = 13 bytes for frame 2005.
   370Ôćĺ */
   371Ôćĺfunction synthesizeOsrsAnimFrameBlob(baseId: number, x: number, y: number, z: number, translateX: number): Uint8Array {
   372Ôćĺ    const bytes: number[] = [];
   373Ôćĺ    // u16 frameLength = 5
   374Ôćĺ    bytes.push((5 >> 8) & 0xff, 5 & 0xff);
   375Ôćĺ
   376Ôćĺ    // group 0 (OP_BASE): flags=0x7, X=x, Y=y, Z=z
   377Ôćĺ    bytes.push(0x7);
   378Ôćĺ    psmartSigned(bytes, x);
   379Ôćĺ    psmartSigned(bytes, y);
   380Ôćĺ    psmartSigned(bytes, z);
   381Ôćĺ
   382Ôćĺ    // group 1 (OP_TRANSLATE): flags=0x1, X=translateX
   383Ôćĺ    bytes.push(0x1);
   384Ôćĺ    psmartSigned(bytes, translateX);
   385Ôćĺ
   386Ôćĺ    // groups 2, 3, 4: flags=0
   387Ôćĺ    bytes.push(0, 0, 0);
   388Ôćĺ
   389Ôćĺ    // u8 delay = 2
   390Ôćĺ    bytes.push(2);
   391Ôćĺ
   392Ôćĺ    // trailing u16 baseId
   393Ôćĺ    bytes.push((baseId >> 8) & 0xff, baseId & 0xff);
   394Ôćĺ
   395Ôćĺ    return new Uint8Array(bytes);
   396Ôćĺ}

   398Ôćĺ/**
   399Ôćĺ * Synthesize an OSRS seq config blob.
   400Ôćĺ *
   401Ôćĺ * Layout (matching `OsrsSeqType.decode` byte-for-byte):
   402Ôćĺ *
   403Ôćĺ *   opcode 1 (frames):  u8 frameCount, then per frame: u16 frameId, u16 iframeId, u16 delay
   404Ôćĺ *   opcode 250:         gjstr debugname
   405Ôćĺ *   opcode 0:           end
   406Ôćĺ *
   407Ôćĺ * For 1 frame pointing to OSRS anim `frameId` with delay=2, debugname=<debugname>:
   408Ôćĺ *   bytes: 1 (opcode), 1 (frameCount), 2 (frameId), 2 (iframeId=65535Ôćĺ-1), 2 (delay), 1 (opcode 250), N (debugname chars) + 1 (terminator), 1 (opcode 0)
   409Ôćĺ */

   411Ôćĺ    const bytes: number[] = [];
   412Ôćĺ    // opcode 1: frames
   413Ôćĺ    bytes.push(1);
   414Ôćĺ    bytes.push(1);                                          // frameCount = 1
   415Ôćĺ    bytes.push((frameId >> 8) & 0xff, frameId & 0xff);     // frameId
   416Ôćĺ    bytes.push(0xff, 0xff);                                 // iframeId = 65535 Ôćĺ -1

   425Ôćĺ

   427Ôćĺ    bytes.push(0);
   428Ôćĺ

   515Ôćĺ/**
   516Ôćĺ * Helper: write a single param entry inside an opcode 249 block.
   517Ôćĺ *
   518Ôćĺ * Layout (per `ParamHelper.decodeParams`):
   519Ôćĺ *   u24 key (3 bytes BE)
   520Ôćĺ *   u1 isString (1 byte ÔÇö 0 for int, 1 for string)
   521Ôćĺ *   if isString: gjstr value
   522Ôćĺ *   else:        i32 value (4 bytes BE, signed)
   523Ôćĺ */
   524Ôćĺfunction writeParam(out: number[], key: number, isString: boolean, value: number | string): void {
   525Ôćĺ    // u24 key (3 bytes BE)
   526Ôćĺ    out.push((key >> 16) & 0xff);
   527Ôćĺ    out.push((key >> 8) & 0xff);
   528Ôćĺ    out.push(key & 0xff);

   533Ôćĺ        for (let i = 0; i < s.length; i++) {

   545Ôćĺ}
